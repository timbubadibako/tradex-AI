```python
class TradingSandbox:
    def __init__(self, idr, coin_holdings, baseline=0.5):
        self.balance_idr = idr
        self.holdings = coin_holdings
        self.baseline = baseline
        # ...
    def get_net_pnl_idr(self, current_price):
        coin_diff = self.holdings - self.baseline
        return self.balance_idr + (coin_diff * current_price)
```
